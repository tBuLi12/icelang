cmake_minimum_required(VERSION 3.14)

set(CMAKE_CXX_COMPILER clang++)
set(CMAKE_C_COMPILER clang)


project(Icy VERSION 0.0.1)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED True)


include(FetchContent)
FetchContent_Declare(
  googletest
  URL https://github.com/google/googletest/archive/03597a01ee50ed33e9dfd640b249b4be3799d395.zip
)

set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
FetchContent_MakeAvailable(googletest)

enable_testing()

include_directories(${PROJECT_SOURCE_DIR}/src/lexer)
include_directories(${PROJECT_SOURCE_DIR}/src/logs)

add_executable(
  lexerTest
  ${PROJECT_SOURCE_DIR}/tests/lexer/testLexer.cpp
  ${PROJECT_SOURCE_DIR}/src/lexer/lexer.cpp
  ${PROJECT_SOURCE_DIR}/src/logs/logs.cpp
)

add_executable(
  logsTests
  ${PROJECT_SOURCE_DIR}/tests/logs/logsTests.cpp
  ${PROJECT_SOURCE_DIR}/src/logs/logs.cpp
)

target_link_libraries(
  lexerTest
  GTest::gtest_main
)

target_link_libraries(
  logsTests
  GTest::gtest_main
)

include(GoogleTest)
gtest_discover_tests(lexerTest)
gtest_discover_tests(logsTests)
